✅ 1. دوال إنشاء وتحويل المصفوفات
1) array()

إنشاء مصفوفة.

$arr = array("A", "B", "C");

2) []

إنشاء مصفوفة بطريقة مختصرة.

$arr = ["A", "B", "C"];

3) array_fill()

ملء مصفوفة بعدد عناصر بقيمة معينة.

$arr = array_fill(0, 5, "Hi");
// ["Hi","Hi","Hi","Hi","Hi"]

4) array_fill_keys()

ملء مصفوفة باستخدام قائمة مفاتيح.

$keys = ["name", "age"];
$arr = array_fill_keys($keys, "unknown");
// ["name"=>"unknown", "age"=>"unknown"]

5) range()

إنشاء مصفوفة أرقام أو حروف متتالية.

$nums = range(1, 5); // [1,2,3,4,5]
$chars = range('A', 'E'); // ['A','B','C','D','E']

✅ 2. دوال طباعة و فحص المصفوفات
6) print_r()

عرض مصفوفة بطريقة واضحة.

print_r($arr);

7) var_dump()

عرض نوع البيانات أيضًا.

var_dump($arr);

8) count() / sizeof()

عدد العناصر.

count([1,2,3]); // 3

9) is_array()

يفحص هل المتغير مصفوفة.

is_array($x);

✅ 3. دوال البحث داخل المصفوفات
10) in_array()

البحث عن قيمة داخل مصفوفة.

in_array("A", ["A","B"]); // true

11) array_search()

إرجاع المفتاح عند العثور على قيمة.

array_search("B", ["A","B","C"]); // 1

12) array_key_exists()

التأكد من وجود مفتاح.

array_key_exists("name", $arr);

13) array_keys()

إرجاع جميع المفاتيح.

array_keys(["a"=>1, "b"=>2]); // ["a","b"]

14) array_values()

إرجاع جميع القيم.

array_values(["a"=>1, "b"=>2]); // [1,2]

✅ 4. دوال الترتيب (Sorting)
15) sort()

ترتيب القيم تصاعديًا — بدون حفظ المفاتيح.

$arr = [3,1,2];
sort($arr); // [1,2,3]

16) rsort()

تنازليًا.

17) asort()

ترتيب القيم مع الاحتفاظ بالمفاتيح.

asort(["b"=>3, "a"=>1]);
// ["a"=>1, "b"=>3]

18) arsort()

تنازليًا مع المفاتيح.

19) ksort()

ترتيب حسب المفاتيح.

20) krsort()

تنازليًا حسب المفاتيح.

21) usort()

الترتيب باستخدام دالة مخصصة.

usort($arr, fn($a,$b) => $a <=> $b);

✅ 5. دوال الإضافة والحذف والتعديل
22) array_push()

إضافة عناصر لنهاية المصفوفة.

$arr = [1,2];
array_push($arr, 3,4); // [1,2,3,4]

23) array_pop()

حذف آخر عنصر.

24) array_unshift()

إضافة لبداية المصفوفة.

25) array_shift()

حذف أول عنصر.

26) array_splice()

حذف / استبدال جزء من مصفوفة.

$arr = [1,2,3,4];
array_splice($arr, 1, 2); 
// يحذف 2 و 3

27) array_slice()

قص جزء من المصفوفة بدون تعديل الأصل.

array_slice([1,2,3,4], 1, 2); // [2,3]

✅ 6. دوال الدمج والتحويل
28) array_merge()

دمج مصفوفتين.

29) array_merge_recursive()

دمج مع دمج القيم المتشابهة داخل مصفوفات فرعية.

30) array_combine()

دمج مصفوفتين (الأولى مفاتيح – الثانية قيم).

array_combine(["a","b"], [1,2]); // ["a"=>1,"b"=>2]

31) array_join()/implode()

تحويل مصفوفة نصوص إلى نص واحد.

implode("-", ["A","B","C"]); // A-B-C

32) explode()

تحويل نص إلى مصفوفة.

explode(",", "A,B,C");

✅ 7. دوال العمليات الحسابية على المصفوفات
33) array_sum()

مجموع العناصر.

34) array_product()

حاصل ضرب العناصر.

✅ 8. دوال الفلترة و التحويل (Callback functions)
35) array_filter()

تصفية المصفوفة.

array_filter([1,2,3,4], fn($n)=>$n%2==0);
// [2,4]

36) array_map()

تطبيق دالة على كل عنصر.

array_map(fn($x)=>$x*2, [1,2,3]); // [2,4,6]

37) array_walk()

تعديل كل عنصر مع إمكانية تعديل الأصل.

38) array_reduce()

تنفيذ عملية تراكمية (مثل جمع).

✅ 9. دوال مقارنة المصفوفات
39) array_diff()

إرجاع العناصر الموجودة في الأولى وغير الموجودة في الثانية.

40) array_diff_assoc()

الفرق حسب القيمة + المفتاح.

41) array_intersect()

إرجاع العناصر المشتركة.

42) array_intersect_assoc()

المشتركة حسب القيمة + المفتاح.

✅ 10. دوال المفاتيح والقيم
43) array_flip()

عكس المفاتيح والقيم.

44) array_reverse()

عكس الترتيب.

45) array_unique()

إزالة العناصر المكررة.
